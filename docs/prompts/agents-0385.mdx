```bash

 -H "Authorization: Bearer $OPENAI_API_KEY" \\

   "messages":[
     {"role":"system","content":"Classify task into {sql, web, writer, code, images}. Return JSON {tool, reason}."},
     {"role":"user","content":"<your input here>"}
   ]
 }'
```
